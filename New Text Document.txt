<script>
/* Simple reminder prototype */
const LS_KEY = 'tablet_reminders_v1';
let reminders = JSON.parse(localStorage.getItem(LS_KEY) || '[]');

const el = id => document.getElementById(id);
const listEl = el('list');
const statusEl = el('status');

function save(){ localStorage.setItem(LS_KEY, JSON.stringify(reminders)); render(); }

function add(title, when, repeat){
  reminders.push({
    id: Date.now().toString(36),
    title: title,
    when: when,
    repeat: repeat
  });
  reminders.sort((a,b)=> new Date(a.when)-new Date(b.when));
  save();
}

function remove(id){
  reminders = reminders.filter(r=> r.id !== id);
  save();
}

function edit(id, newTitle, newWhen, newRepeat){
  const r = reminders.find(x=>x.id===id);
  if(!r) return;
  r.title = newTitle; r.when = newWhen; r.repeat = newRepeat;
  save();
}

function nextReminder(){
  const now = Date.now();
  for(const r of reminders){
    const t = new Date(r.when).getTime();
    if(t >= now) return r;
  }
  return null;
}

function render(){
  listEl.innerHTML = '';
  if(reminders.length===0){
    listEl.innerHTML = '<div class="small">No reminders yet — add one above.</div>';
  } else {
    for(const r of reminders){
      const div = document.createElement('div'); div.className='rem';
      const left = document.createElement('div');
      left.innerHTML = `<div style="font-weight:600">${r.title}</div><div class="small">${new Date(r.when).toLocaleString()} · ${r.repeat}</div>`;
      const right = document.createElement('div'); right.style.display='flex'; right.style.alignItems='center';
      right.innerHTML = `<div class="time">${timeUntil(r.when)}</div>`;
      const actions = document.createElement('div'); actions.className='actions';
      const del = document.createElement('button'); del.textContent='Delete'; del.onclick = ()=> remove(r.id);
      actions.appendChild(del);
      right.appendChild(actions);
      div.appendChild(left); div.appendChild(right);
      listEl.appendChild(div);
    }
  }
  const nr = nextReminder();
  statusEl.textContent = nr ? `Next: ${nr.title} at ${new Date(nr.when).toLocaleString()}` : 'No upcoming reminders';
}

function timeUntil(iso){
  const t = new Date(iso).getTime() - Date.now();
  if(t <= 0) return 'Now / overdue';
  const mins = Math.floor(t/60000);
  if(mins < 60) return `${mins}m`;
  const hrs = Math.floor(mins/60);
  if(hrs < 24) return `${hrs}h`;
  const days = Math.floor(hrs/24);
  return `${days}d`;
}

/* Form handling */
el('addBtn').addEventListener('click', ()=>{
  const title = el('title').value.trim();
  const when = el('date').value;
  const repeat = el('repeat').value;
  if(!title || !when){ alert('Fill title and date/time'); return; }
  add(title, when, repeat);
  el('title').value=''; el('date').value=''; el('repeat').value='none';
});

/* Seed sample if empty (so judges see something) */
if(reminders.length===0){
  reminders = [
    {id:'seed1',title:'Morning tablet',when:new Date(Date.now()+3600*1000).toISOString().slice(0,16),repeat:'daily'},
    {id:'seed2',title:'Evening tablet',when:new Date(Date.now()+5*3600*1000).toISOString().slice(0,16),repeat:'none'}
  ];
  save();
} else {
  render();
}

/* Periodic check (visual) */
setInterval(render, 10000);
render();
</script>
